cb_good_shortage_yes = {
	exists = market
	market = {
		$MARKET_GOOD$ = {
			market_goods_has_goods_shortage = yes
			market_goods_buy_orders > 0
		}
	}
}

cb_good_shortage_no = {
	exists = market
	market = {
		$MARKET_GOOD$ = { market_goods_has_goods_shortage = no }
	}
}

cb_good_expensive_yes = {
	exists = market
	market = {
		$MARKET_GOOD$ = {
			is_market_goods_expensive = yes
			market_goods_buy_orders > 0
		}
	}

cb_state_has_free_workforce = {
	OR = {
		state_unemployment_rate >= 0.02
		AND = {
			any_scope_pop = {
				is_pop_type = peasants
				OR = {
					pop_employment_building_group =	bg_subsistence_agriculture
					pop_employment_building_group =	bg_subsistence_ranching
				}
			}
			any_scope_building = {
				is_subsistence_building = yes
				level > 0
				occupancy >= 0.02
			}
		}
	}
}

cb_building_for_emptiness_check = {
	level > 0
	NOR = {
		is_subsistence_building = yes
		is_building_type = building_urban_center
		is_building_type = building_trade_center
		is_building_type = building_railway
		is_building_type = building_conscription_center
		is_building_type = building_barracks
		is_building_type = building_naval_base
	}
}

cb_state_has_multiple_empty_buildings = {
	any_scope_building = {
		cb_building_for_emptiness_check = yes
	}
	OR = {
		NOT = {
			any_scope_building = {
				cb_building_for_emptiness_check = yes
				occupancy >= 0.90
			}
		}
		any_scope_building = {
			count >= 3
			cb_building_for_emptiness_check = yes
			occupancy < 0.60
		}
	}
}

cb_building_is_upgradable = {
	is_under_construction = no
	is_buildable = yes
	OR = {
		level <= 0
		AND = {
			level >= 1
			building_has_goods_shortage = no
			occupancy >= 0.90
			#cash_reserves_ratio >= 0.25
		}
	}
}

cb_should_build_port_critical = {
	has_technology_researched = navigation
	any_scope_state = {
		has_coast = yes
	}
	OR = {
		modifier:country_convoys_capacity_add < cb_convoys_critical
		any_scope_state = {
			has_coast = yes
			is_isolated_from_market = yes
			NOT = {
				has_building = building_port
			}
		}
		AND = {
			OR = {
				NOT = {
					has_technology_researched = railways
				}
				cb_good_shortage_yes = { MARKET_GOOD = mg:engines }
			}
			any_scope_state = {
				has_coast = yes
				infrastructure < cb_infrastructure_critical
			}
		}
	}
	ct_is_budget_acceptable = yes
}

cb_should_build_port_important = {
	has_technology_researched = navigation
	any_scope_state = {
		has_coast = yes
	}
	OR = {
		modifier:country_convoys_capacity_add < cb_convoys_important
		AND = {
			OR = {
				NOT = {
					has_technology_researched = railways
				}
				cb_good_shortage_yes = { MARKET_GOOD = mg:engines }
			}
			any_scope_state = {
				has_coast = yes
				infrastructure < cb_infrastructure_important
			}
		}
		any_country = {
			OR = {
				is_subject_of = root
				is_non_customs_union_subject = no
				AND = {
					leads_customs_union = no
					OR = {
						has_diplomatic_pact = {
							who = root
							type = customs_union
						}
						save_temporary_scope_as = current_country
						root = {
							has_diplomatic_pact = {
								who = scope:current_country
								type = customs_union
							}
						}
					}
				}
			}
			any_scope_state = {
				has_coast = yes
				infrastructure >= infrastructure_usage
				market_access < 1
			}
		}
	}
	ct_is_budget_acceptable = yes
}

cb_should_build_port_profitable = {

	any_scope_state = {

	}
